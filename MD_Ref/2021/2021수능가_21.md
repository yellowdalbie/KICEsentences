\[
\begin{array}{l}
\text { (가) } a_{2 n}=a_{2} \times a_{n}+1 \\
\text { (나) } a_{2 n+1}=a_{2} \times a_{n}-2
\end{array}
\]
$$a_{8}-a_{15}=63 \text { 일 때, } \frac{a_{8}}{a_{1}} \text { 의 값은? [4점] }$$

(2) 92
